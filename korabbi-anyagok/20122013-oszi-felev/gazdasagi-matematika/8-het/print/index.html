set.seed(2)
df <- data.frame(rnd = runif(90) * 1000)
df$mday <- 1:30
df$wday <- rep(1:7, 15)[1:90]
df$mtrend <- seq(60000, 2000, -2000)
df$wtrend <- 0
df$wtrend[which(df$wday == 1)] <- 20000
df$wtrend[which(df$wday == 2)] <- 30000
df$wtrend[which(df$wday == 3)] <- 40000
df$wtrend[which(df$wday == 4)] <- 30000
df$wtrend[which(df$wday == 5)] <- 20000
df$atm <- df$wtrend + df$mtrend + df$rnd
plot(df$atm)
plot(df$atm, type = 'l')
